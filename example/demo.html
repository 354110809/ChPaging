动态DEMO

代码:

//html
< script src="./jquery.2.1.2.min.js">< /script>
< script src="../chPaging.1.0.1.js">< /script>
< ul id="demo1_list" class="demo_list">< /ul>
< div id="demo1_page" class="demo_page">< /div>

//css
    .demo_list{
        list-style: none;
        margin-top: 30px;
    }
    .demo_list li{
        height: 30px;
    }
    .demo_page span{
        display: inline-block;
        margin:0 10px;
    }
    .demo_page a{
        display: inline-block;
        margin: 0 5px;
    }
    .demo_page a.ChPaging_choose{
        color: #f00;
    }
    .demo_page input{
        width: 30px;
    }
    .demo_page button{
        width: 40px;
        height: 21px;
        line-height: 20px;
    }

//js
< script>
    var demoYP1 = new ChPaging("#demo1_page",{
            limit : 5
            ,viewNumber : false
            ,xhr : {
                url : "./test.json"
                ,data : {
                    limit : 5,
                    current : 1
                }
            }
            ,xhrSuccess : function(data){
                return  {
                    data : data.data
                    ,count : data.count
                };
            }
            ,operReady : function (msg) {
                demoYP1.set({//设置请求参数
                    xhr : {
                        data : {
                            current : msg.current
                        }
                    }
                });
            }
            ,operFinsh : function(data){
                $("#demo1_list").html("");
                for(var i = 0; i < data.length; i++){
                    var index = (demoYP1.get("start") + i);
                    $("#demo1_list").append("< li>第"+ index + "条。模拟数据.手动变化"+ index + index + index + index  + data[i]+"< /li>")
                }
            }
        });
< /script>
