//html
< script src="http://libs.baidu.com/jquery/2.1.2/jquery.min.js">< /script>
< script src="../yx-paging.1.0.1.js">< /script>
< ul id="demo1_list" class="demo_list">< /ul>
< div id="demo1_page" class="demo_page">< /div>

//css
    .demo_list{
        list-style: none;
        margin-top: 30px;
    }
    .demo_list li{
        height: 30px;
    }
    .demo_page span{
        display: inline-block;
        margin:0 10px;
    }
    .demo_page a{
        display: inline-block;
        margin: 0 5px;
    }
    .demo_page a.yxPaging_choose{
        color: #f00;
    }
    .demo_page input{
        width: 30px;
    }
    .demo_page button{
        width: 40px;
        height: 21px;
        line-height: 20px;
    }

//js
< script>
    var demoYP1 = new YxPaging("#demo1_page",{
        limit : 5
        ,data : data
        ,xhr : {
            url : "../test.json"
            ,data : {
                start : 0,
                end : 10
            }
        }
        ,xhrSuccess : function(data){
            return  {
                data : data.data
                ,count : data.count
            };
        }
        ,reloadPage : function(msg,data){
            console.log(msg,data);
            $("#demo1_list").html("");
            for(var i = 0; i < data.length; i++){
                var index = (msg.start+i) + 1;
                $("#demo1_list").append("< li>第"+ index + "条。模拟数据.手动变化"+ index + index + index + index  + data[i]+"< /li>")
            }
        }
    })
< /script>
